This task addresses two issues:

1) In common/multichannel/Bundle.ValidateNew() it is possible
   to identify the system channel using:
       _, isSys := b.ConsortiumsConfig()
   this can be used to refine validateMigrationStep() such that
   it deals more accurately with the migration-state transitions
   on the system vs. standard channels.

2) In addition, prevent user from adding ConsortiumsConfig() to
   standard channels. This will protect multichannel.Registrar
   from blowing up on next initialization. Explanation:

   - Looking at the code in multichannel.Registrar, we see that
       _, ok := ledgerResources.ConsortiumsConfig()
   is used to identify the system channel. If two system channels
   are identified, the code panics.

   - Now, in Bundle.ValidateNew(), currently there is no mechanism
   to prevent a user (orderer admin) from adding a ConsortiumsConfig()
   to a standard channel. If a user does that,  multichannel.Registrar
   will blow up in the next initialization.

package main

import (
	"fmt"
	"github.com/hyperledger/fabric/protos/orderer"
	"strconv"
)

type Context struct {
	comparator string
	number     int
}

func (c Context) equal(actual Context) bool {
	switch c.comparator {
	case ">=":
		return actual.number >= c.number
	case "<=":
		return actual.number <= c.number
	case ">":
		return actual.number > c.number
	case "<":
		return actual.number < c.number
	case "==":
		fallthrough
	default:
		return actual.number == c.number
	}
}

type State struct {
	ConsensusType string
	MIGState      orderer.ConsensusType_MigrationState
	Context       Context

	next []State
}

func (a State) Equal(b State) bool {
	return a.ConsensusType == b.ConsensusType && a.MIGState == b.MIGState && a.Context.equal(b.Context)
}

type Transitions []State

func (t Transitions) Allowed(from, to State) bool {
	for _, this := range t {
		if !this.Equal(from) {
			continue
		}

		for _, next := range this.next {
			if next.Equal(to) {
				return true
			}
		}

		return false
	}

	return false
}

func (t *Transitions) Add(from, to State) {
	for _, this := range *t {
		if !this.Equal(from) {
			continue
		}

		for _, next := range this.next {
			if next.Equal(to) {
				return
			}
		}

		this.next = append(this.next, to)
		return
	}

	from.next = []State{to}
	*t = append(*t, from)
}

func state(t string, s orderer.ConsensusType_MigrationState, c string) State {
	var comparator string
	for i, char := range c {
		if '0' <= char && char <= '9' {
			comparator = c[:i]
			n, _ := strconv.Atoi(c[i:])
			return State{
				ConsensusType: t,
				MIGState:      s,
				Context:       Context{comparator: comparator, number: n},
			}
		}
	}

	panic("Programming error")
}

func main() {
	var whitelist Transitions
	whitelist.Add(
		state("kafka", orderer.ConsensusType_MIG_STATE_NONE, "0"),
		state("etcdraft", orderer.ConsensusType_MIG_STATE_START, ">0"))

	from := state("kafka", orderer.ConsensusType_MIG_STATE_NONE, "0")
	to := state("kafka", orderer.ConsensusType_MIG_STATE_START, "20")
	fmt.Printf("%+v, %+v, %t\n", from, to, whitelist.Allowed(from, to)) // this should be false

	to = state("etcdraft", orderer.ConsensusType_MIG_STATE_START, "20")
	fmt.Printf("%+v, %+v, %t\n", from, to, whitelist.Allowed(from, to)) // this should be true
}
