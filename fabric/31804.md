func (msp *bccspmsp) getIdentityFromConf(idBytes []byte) (Identity,
bccsp.Key, error) {
// get a cert
cert, err := msp.getCertFromPem(idBytes)
if err != nil
{ return nil, nil, err }

// get the public key in the right format
certPubK, err := msp.bccsp.KeyImport(cert,
&bccsp.X509PublicKeyImportOpts{Temporary: true})

mspId, err := newIdentity(cert, certPubK, msp)
if err != nil { return nil, nil, err }
return mspId, certPubK, nil
}
